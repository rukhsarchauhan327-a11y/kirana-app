    <script>
        let isYearly = false;

        document.getElementById('monthly-tab').addEventListener('click', function() {
            if (isYearly) {
                switchToMonthly();
            }
        });

        document.getElementById('yearly-tab').addEventListener('click', function() {
            if (!isYearly) {
                switchToYearly();
            }
        });

        function switchToMonthly() {
            isYearly = false;
            
            // Update tab styling
            document.getElementById('monthly-tab').classList.add('bg-white', 'text-dark', 'shadow-sm');
            document.getElementById('monthly-tab').classList.remove('text-gray-600');
            document.getElementById('yearly-tab').classList.remove('bg-white', 'text-dark', 'shadow-sm');
            document.getElementById('yearly-tab').classList.add('text-gray-600');
            
            // Show/hide plan sections
            document.getElementById('monthly-plans').classList.remove('hidden');
            document.getElementById('yearly-plans').classList.add('hidden');
            
            // Hide discount badge
            document.getElementById('discount-badge').classList.add('hidden');
        }

        function switchToYearly() {
            isYearly = true;
            
            // Update tab styling
            document.getElementById('yearly-tab').classList.add('bg-white', 'text-dark', 'shadow-sm');
            document.getElementById('yearly-tab').classList.remove('text-gray-600');
            document.getElementById('monthly-tab').classList.remove('bg-white', 'text-dark', 'shadow-sm');
            document.getElementById('monthly-tab').classList.add('text-gray-600');
            
            // Show/hide plan sections
            document.getElementById('yearly-plans').classList.remove('hidden');
            document.getElementById('monthly-plans').classList.add('hidden');
            
            // Show discount badge
            document.getElementById('discount-badge').classList.remove('hidden');
        }

        function toggleFAQ(index) {
            const content = document.getElementById(`faq-content-${index}`);
            const icon = document.getElementById(`faq-icon-${index}`);
            
            if (content.classList.contains('hidden')) {
                content.classList.remove('hidden');
                icon.style.transform = 'rotate(180deg)';
            } else {
                content.classList.add('hidden');
                icon.style.transform = 'rotate(0deg)';
            }
        }
    </script>
