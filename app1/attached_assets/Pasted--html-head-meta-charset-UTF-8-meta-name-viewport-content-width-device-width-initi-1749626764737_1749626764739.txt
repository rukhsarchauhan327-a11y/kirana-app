    <script>
        const addToCartBtn = document.getElementById('add-to-cart-btn');
        const cartModal = document.getElementById('cart-modal');
        const closeModal = document.getElementById('close-modal');
        const cancelBtn = document.getElementById('cancel-btn');
        const quantityInput = document.getElementById('quantity-input');
        const priceInput = document.getElementById('price-input');
        const unitSelect = document.getElementById('unit-select');
        const calculatedQuantity = document.getElementById('calculated-quantity');
        const calculatedPrice = document.getElementById('calculated-price');
        const quantitySection = document.getElementById('quantity-section');
        const priceSection = document.getElementById('price-section');
        const inputMethodRadios = document.querySelectorAll('input[name="inputMethod"]');

        const unitPrice = 110;
        const maxStock = 25.5;

        addToCartBtn.addEventListener('click', () => {
            cartModal.classList.remove('hidden');
        });

        closeModal.addEventListener('click', () => {
            cartModal.classList.add('hidden');
        });

        cancelBtn.addEventListener('click', () => {
            cartModal.classList.add('hidden');
        });

        cartModal.addEventListener('click', (e) => {
            if (e.target === cartModal) {
                cartModal.classList.add('hidden');
            }
        });

        inputMethodRadios.forEach(radio => {
            radio.addEventListener('change', (e) => {
                if (e.target.value === 'quantity') {
                    quantitySection.classList.remove('hidden');
                    priceSection.classList.add('hidden');
                    updateFromQuantity();
                } else {
                    quantitySection.classList.add('hidden');
                    priceSection.classList.remove('hidden');
                    updateFromPrice();
                }
            });
        });

        quantityInput.addEventListener('input', updateFromQuantity);
        unitSelect.addEventListener('change', updateFromQuantity);
        priceInput.addEventListener('input', updateFromPrice);

        function updateFromQuantity() {
            let qty = parseFloat(quantityInput.value) || 0;
            const unit = unitSelect.value;
            
            if (unit === 'gm') {
                qty = qty / 1000;
            }
            
            const total = qty * unitPrice;
            
            let displayQty = qty;
            let displayUnit = 'Kg';
            
            if (qty < 1) {
                displayQty = qty * 1000;
                displayUnit = 'gm';
            }
            
            calculatedQuantity.textContent = `${displayQty.toFixed(displayQty < 1 ? 0 : 3)} ${displayUnit}`;
            calculatedPrice.textContent = `₹${total.toFixed(2)}`;
        }

        function updateFromPrice() {
            const price = parseFloat(priceInput.value) || 0;
            const qty = price / unitPrice;
            
            let displayQty = qty;
            let displayUnit = 'Kg';
            
            if (qty < 1) {
                displayQty = qty * 1000;
                displayUnit = 'ml';
            }
            
            calculatedQuantity.textContent = `${displayQty.toFixed(displayQty < 1 ? 0 : 3)} ${displayUnit}`;
            calculatedPrice.textContent = `₹${price.toFixed(2)}`;
        }

        updateFromQuantity();
    </script>
